```typescript
interface ChatMessage {
  role: "user" | "bot";
  content: string;
  timestamp: string;
}

interface ChatStorage {
  messages: ChatMessage[];
  lastActive: string;
}

const STORAGE_KEY = "chat-history";
const MAX_MESSAGES = 50;
const TTL_MS = 24 * 60 * 60 * 1000; // 24 hours
```

**2. Add save function:**

```typescript
function saveChatHistory(msgs: ChatMessage[]): void {
  const data: ChatStorage = {
    messages: msgs.slice(-MAX_MESSAGES),
    lastActive: new Date().toISOString(),
  };
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
  } catch {
    // Silently fail -- localStorage may be full or disabled
  }
}
```

**3. Add load function:**

```typescript
function loadChatHistory(): ChatMessage[] | null {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (!raw) return null;
    const data: ChatStorage = JSON.parse(raw);
    const elapsed = Date.now() - new Date(data.lastActive).getTime();
    if (elapsed > TTL_MS) {
      localStorage.removeItem(STORAGE_KEY);
      return null;
    }
    return data.messages;
  } catch {
    localStorage.removeItem(STORAGE_KEY);
    return null;
  }
}
```
