**1. Refactor `src/scripts/animations.ts`:**

Replace the entire file. The new version:
- Exports `initAnimations` as an `async` function (not a local function called inline)
- Exports `cleanupAnimations` (unchanged logic)
- Moves all GSAP imports inside `initAnimations()` as dynamic `import()` calls
- Checks `prefers-reduced-motion` BEFORE importing GSAP — if reduced motion is active, set all `[data-animate]` elements to `opacity: 1` and `transform: none`, then return WITHOUT importing GSAP at all (per D-05)
- Removes the `document.addEventListener` calls at the bottom (the loader moves to BaseLayout)

New file structure:
```typescript
let ctx: any = null;

export async function initAnimations() {
  // D-05: completely disable animations on prefers-reduced-motion
  if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
    document.querySelectorAll('[data-animate]').forEach(el => {
      (el as HTMLElement).style.opacity = '1';
      (el as HTMLElement).style.transform = 'none';
    });
    return;
  }

  // Dynamic imports — Vite code-splits these into separate chunks
  const { gsap } = await import('gsap');
  const { ScrollTrigger } = await import('gsap/ScrollTrigger');
  const { SplitText } = await import('gsap/SplitText');

  gsap.registerPlugin(ScrollTrigger, SplitText);

  ctx = gsap.context(() => {
    // Section reveals (same animation config as current)
    gsap.utils.toArray<HTMLElement>('[data-animate="section"]').forEach((el) => {
      gsap.from(el, {
        opacity: 0, y: 24, duration: 0.8, ease: 'power2.out',
        scrollTrigger: { trigger: el, start: 'top 85%', once: true },
      });
    });

    // Stagger items (same animation config as current)
    gsap.utils.toArray<HTMLElement>('[data-animate-container="stagger"]').forEach((container) => {
      const items = container.querySelectorAll('[data-animate="stagger-item"]');
      if (items.length === 0) return;
      gsap.from(items, {
        opacity: 0, y: 24, duration: 0.6, stagger: 0.1, ease: 'power2.out',
        scrollTrigger: { trigger: container, start: 'top 85%', once: true },
      });
    });

    // SplitText display headings (same animation config as current)
    gsap.utils.toArray<HTMLElement>('[data-animate="split-text"]').forEach((el) => {
      SplitText.create(el, {
        type: 'lines', mask: 'lines', autoSplit: true,
        onSplit(self: any) {
          return gsap.from(self.lines, {
            y: '100%', opacity: 0, duration: 0.7, stagger: 0.12, ease: 'power2.out',
          });
        },
      });
    });
  });
}

export function cleanupAnimations() {
  ctx?.revert();
  ctx = null;
}
```

**2. Update `src/layouts/BaseLayout.astro`:**

Replace the existing `<script>` block (lines 100-103) with a new inline script that:
- On `astro:page-load`, dynamically imports `initAnimations` from `../scripts/animations.ts` and calls it
- On `astro:before-preparation`, dynamically imports `cleanupAnimations` and calls it
- Adds a 3-second timeout fallback that reveals any still-hidden `[data-animate]` elements (in case GSAP fails to load — Safari import bug protection per RESEARCH Pitfall 6)

New script block:
```astro
<script>
  document.addEventListener('astro:page-load', async () => {
    const { initAnimations } = await import('../scripts/animations.ts');
    await initAnimations();
  });

  document.addEventListener('astro:before-preparation', async () => {
    const { cleanupAnimations } = await import('../scripts/animations.ts');
    cleanupAnimations();
  });

  // Fallback: reveal content if GSAP fails to load after 3 seconds
  setTimeout(() => {
    document.querySelectorAll('[data-animate]').forEach(el => {
      const computed = getComputedStyle(el);
      if (computed.opacity === '0') {
        (el as HTMLElement).style.opacity = '1';
        (el as HTMLElement).style.transform = 'none';
      }
    });
  }, 3000);
</script>
```
